<let_comando> ::= "let" ID "=" <expr>
<expr> ::= <termo> <K>
<termo> ::= INT | FLOAT
<K> ::= (POW | PLUS | MINUS | MUL | DIV ) <termo> <K>
<K> ::= ε


PLUS é o +
MUL é o *
DIV é o /
MINUS é o -
POW é o ^
INT é um inteiro qualquer
FLOAT é um float qualquer
ID é qualquer identificador
LET é a palavra reservada let
